import { useState } from 'react';
import { Meta, Story, Canvas, ArgsTable } from '@storybook/addon-docs/blocks';
import { RadioButtonWithLabel } from './RadioButtonWithLabel.js';

<Meta title="Components/RadioButtonWithLabel" />

export const commonArgTypes = {
  onToggle: { action: 'onToggle' },
};

<Canvas>
  <Story
    name="Default"
    args={{
      id: '123',
      name: 'my-checkbox',
      checked: false,
      disabled: false,
      label: 'Click me',
    }}
    argTypes={commonArgTypes}
  >
    {(args) => {
      const [checked, setChecked] = useState(false);
      return (
        <RadioButtonWithLabel
          {...args}
          checked={checked}
          onToggle={(event) => {
            args.onToggle(event);
            setChecked(!checked);
          }}
        />
      );
    }}
  </Story>
</Canvas>

<ArgsTable story="Default" />
