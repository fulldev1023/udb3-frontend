## Default

<Canvas>
  <Story
    name="Default"
    args={{
      currentPage: 1,
      totalItems: 50,
      perPage: 10,
      limitPages: 5,
      prevText: 'Previous',
      nextText: 'Next',
    }}
  >
    {(args) => {
      const [currentPage, setCurrentpage] = useState(1);
      return (
        <Pagination
          {...args}
          currentPage={currentPage}
          onChangePage={(newPage) => {
            args.onChangePage();
            setCurrentpage(newPage);
          }}
        />
      );
    }}
  </Story>
