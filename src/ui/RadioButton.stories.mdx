import { useState } from 'react';
import { Meta, Story, Canvas, ArgsTable } from '@storybook/addon-docs/blocks';
import { RadioButton, RadioButtonTypes } from './RadioButton';

<Meta title="Components/RadioButton" component={RadioButton} />

export const argTypes = {
  type: {
    control: {
      type: 'select',
      options: Object.values(RadioButtonTypes),
    },
  },
};

<Canvas>
  <Story
    name="Default"
    args={{
      id: '123',
      name: 'my-radiobutton',
      disabled: false,
      type: RadioButtonTypes.RADIO,
      onChange: () => {},
    }}
    argTypes={argTypes}
  >
    {(args) => {
      const [checked, setChecked] = useState(false);
      return (
        <RadioButton
          {...args}
          checked={checked}
          onChange={(e) => {
            setChecked(e.target.checked);
            args.onChange(e.target.checked);
          }}
        />
      );
    }}
  </Story>
</Canvas>

<ArgsTable story="Default" />
