import { Meta, Story, Canvas, ArgsTable } from '@storybook/addon-docs/blocks';
import { RadioButtonWithLabel } from './RadioButtonWithLabel';
import { RadioButtonTypes } from './RadioButton';
import { useState } from 'react';

<Meta
  title="Components/RadioButtonWithLabel"
  component={RadioButtonWithLabel}
/>

export const argTypes = {
  type: {
    control: {
      type: 'select',
      options: [...Object.values(RadioButtonTypes)],
    },
  },
};

<Canvas>
  <Story
    name="Default"
    args={{
      id: '123',
      name: 'my-checkbox',
      disabled: false,
      label: 'Online',
      type: RadioButtonTypes.RADIO,
      onChange: () => {},
    }}
    argTypes={argTypes}
  >
    {(args) => {
      const [checked, setChecked] = useState(false);
      return (
        <RadioButtonWithLabel
          {...args}
          checked={checked}
          onChange={(e) => {
            setChecked(e.target.checked);
            args.onChange(e.target.checked);
          }}
        />
      );
    }}
  </Story>
</Canvas>

<ArgsTable story="Default" />
