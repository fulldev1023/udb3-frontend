<Meta
  title="Components/pub-modal/pub-modal-content"
  component={PubModalContent}
/>

export const Template = (args, { argTypes }) => {
  return {
    props: Object.keys(argTypes),
    data() {
      return {
        isVisible: false,
      };
    },
    components: { PubModalContent, PubButton },
    methods: {
      handleClick() {
        this.isVisible = true;
      },
      handleHide() {
        this.isVisible = false;
      },
    },
    template:
      '<div><pub-button @click="handleClick">Open modal</pub-button><pub-modal-content v-bind="$props" :visible="isVisible" @hide="handleHide">Modal content</pub-modal-content></div>',
  };
};

**
This component is a wrapper around the [b-modal](https://bootstrap-vue.org/docs/components/modal) component from bootstrap-vue.
