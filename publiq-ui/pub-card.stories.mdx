import { Meta, Story, Canvas, ArgsTable } from '@storybook/addon-docs/blocks';
import PubCard from './pub-card.vue';
import PubCardText from './pub-card-text.vue';

<Meta title="Components/pub-card" component={PubCard} />

export const TemplateCard = (args, { argTypes }) => {
  return {
    props: Object.keys(argTypes),
    components: { PubCard },
    template: '<pub-card v-bind="$props">Card content</pub-card>',
  };
};

export const TemplateCardWithCardText = (args, { argTypes }) => {
  return {
    props: Object.keys(argTypes),
    components: { PubCard, PubCardText },
    template:
      '<pub-card v-bind="$props">Card content <pub-card-text v-bind="$props">Card footer text</pub-card-text></pub-card>',
  };
};
